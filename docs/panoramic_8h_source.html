24 inline std::unique_ptr<Input> getPanoramicCamera()

26 #ifdef _WIN32
27  // for Windows we currently just select the first camera
28  return std::make_unique<OpenCVInput>(0, cv::Size(1280, 720), "webcam360");
29 #else
30  // get vector of video input devices on system
31  auto cameras = OS::Video::getCameras();
32 
33  // list of preferred cameras in priority order
34  static const std::array<std::string, 3> prefCameras{
35  SEE3CAM_DEVICE_NAME, PIXPRO_USB_DEVICE_NAME, WEBCAM360_DEVICE_NAME
36  };
37 
38  int deviceNum = -1, prefCamNum = -1;
39  size_t lowestIndex = prefCameras.size();
40  for (const auto &cam : cameras) {
41  // Iterate through prefCameras looking for matches
42  for (size_t i = 0; i < lowestIndex; i++) {
43  if (cam.second == prefCameras[i]) {
44  deviceNum = cam.first;
45  prefCamNum = i;
46  lowestIndex = i;
47  break;
48  }
49  }
50 
51  // If we matched the most preferred camera then we're done
52  if (prefCamNum == 0) {
53  break;
54  }
55 
56  /*
57  * Even if we didn't match any preferred camera, at least this is a
58  * camera, so use as fallback
59  */
60  if (deviceNum == -1) {
61  deviceNum = cam.first;
62  }
63  }
64  std::cout << "Selected camera #" << deviceNum << ": "
65  << OS::Video::getCameraName(deviceNum) << std::endl;
66 
67  if (deviceNum == -1) {
68  throw std::runtime_error("No camera found");
69  }
70 
71  // SeeCam
72  if(prefCamNum == 0) {
73  auto see3cam = std::make_unique<See3CAM_CU40>("/dev/video" + std::to_string(deviceNum),
74  See3CAM_CU40::Resolution::_1280x720);
75  // Run auto exposure algorithm
76  const cv::Mat bubblescopeMask = See3CAM_CU40::createBubblescopeMask(see3cam->getSuperPixelSize());
77  see3cam->autoExposure(bubblescopeMask);
78  return see3cam;
79  }
80  // PixPro
81  else if(prefCamNum == 1) {
82  return std::make_unique<OpenCVInput>(deviceNum, cv::Size(1440, 1440), "pixpro_usb");
83  }
84  // Default
85  else {
86  return std::make_unique<OpenCVInput>(deviceNum, cv::Size(1280, 720), "webcam360");
87  }
88 #endif
89 }
