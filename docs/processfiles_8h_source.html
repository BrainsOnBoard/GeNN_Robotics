54 /* unwrap an MP4 video */
55 void processmp4(const char *filepathRaw)
56 {

58 
59  // open video file
60  cv::VideoCapture cap(filepath.str());
61 
62  // read in first frame
63  cv::Mat fr;
64  cap >> fr;
65 
66  // matrices to store maps for unwrapping
67  cv::Mat map_x(unwrap_height, unwrap_width, CV_32FC1);
68  cv::Mat map_y(unwrap_height, unwrap_width, CV_32FC1);
69 
70  // matrix to store unwrapped image
71  cv::Mat imunwrap(unwrap_height, unwrap_width, fr.type());
72 
73  // try unwrapping first frame and return if fail
74  if (!processframe(filepath.str().c_str(), imunwrap, fr, map_x, map_y)) {
75  cap.release();
76  return;
77  }
78 
79  // final filename for unwrapped video
80  filesystem::path outfilename = filepath.parent_path() / ("unwrapped_" + filepath.filename());
81 
82  // temporary file name to which we write initially
83  filesystem::path tempfilename = copysound ? filepath.parent_path() / ".TEMP.MP4": outfilename;
84 
85  // start writing to file
86  std::cout << "Saving video to " << outfilename << "..." << std::endl;
87  cv::VideoWriter writer(
88  tempfilename.str(), 0x21, cap.get(CV_CAP_PROP_FPS), imunwrap.size());
89  if (!writer.isOpened()) {
90  std::cerr << "Error: Could not open file for writing" << std::endl;
91  return;
92  }
93 
94  // write first unwrapped frame
95  writer.write(imunwrap);
96 
97  // unwrap successive frames and write to file
98  for (;;) {
99  cap >> fr;
100  if (fr.empty()) {
101  break;
102  }
103 
104  unwrap(imunwrap, fr, map_x, map_y);
105  writer.write(imunwrap);
106  }
107 
108  // dispose of writer and reader when finished
109  cap.release();
110  writer.release();
111 
112  if (copysound) {
113  // attempt to copy audio stream from original to new file with ffmpeg
114  int stat = system((FFMPEG_PATH " -y -i \"" + tempfilename.str() + "\" -i \"" +
115  filepath.str() +
116  "\" -map 0:v -map 1:a -c copy -shortest \"" +
117  outfilename.str() + "\" >/dev/null 2>&1")
118  .c_str());

122

126  */
127  rename(tempfilename.str().c_str(), outfilename.str().c_str());
