23 static radian_t
24 circularMean(const T1 &angles, const T2 &weights)
25 {
26  scalar_t sumCos = 0.0;
27  scalar_t sumSin = 0.0;
28  for (size_t i = 0; i < angles.size(); i++) {
29  sumCos += weights[i] * units::math::cos(angles[i]);
30  sumSin += weights[i] * units::math::sin(angles[i]);
31  }
32 
33  return units::math::atan2(sumSin / angles.size(), sumCos / angles.size());

66 template<size_t numComp>
67 struct WeightSnapshotsDynamic
68 {
69  auto operator()(const cv::Size &unwrapRes,
70  std::vector<int> &bestCols,
71  std::vector<float> &minDifferences)
72  {
73  // Create vector of indices
74  std::vector<size_t> idx(minDifferences.size());
75  std::iota(std::begin(idx), std::end(idx), 0);
76 
77  // Build a heap (only partially sorts)
78  auto comparator = [&minDifferences](const size_t i1, const size_t i2) {
79  return minDifferences[i1] >= minDifferences[i2];
80  };
81  std::make_heap(std::begin(idx), std::end(idx), comparator);
82 
83  // Pop the best numComp indices off the heap into the output array
84  std::array<size_t, numComp> snapshots;
85  for(size_t &s : snapshots) {
86  std::pop_heap(std::begin(idx), std::end(idx), comparator);
87  s = idx.back();
88  idx.pop_back();
89  }
90 
91  // Convert best columns to headings
92  auto colsToHeadings = [&bestCols, &unwrapRes](const size_t s) {
93  return units::make_unit<turn_t>((double) bestCols[s] / (double) unwrapRes.width);
94  };
95  std::array<radian_t, numComp> headings;
96  std::transform(snapshots.cbegin(), snapshots.cend(), headings.begin(),
97  colsToHeadings);
98 
99  // Normalise min differences to be between 0 and 1
100  auto normaliseDiffs = [&minDifferences](const size_t i) {
101  return minDifferences[i] / 255.0f;
102  };
103  std::array<float, numComp> minDifferencesOut;
104  std::transform(snapshots.cbegin(), snapshots.cend(),
105  minDifferencesOut.begin(), normaliseDiffs);
106 
107  // Weights are 1 - min differences
108  const auto diffsToWeights = [](const float f) {
109  return 1.0f - f;
110  };
111  std::array<float, numComp> weights;
112  std::transform(minDifferencesOut.cbegin(), minDifferencesOut.cend(),
113  weights.begin(), diffsToWeights);
114 
115  // Best angle is a weighted cirular mean of headings
116  const radian_t bestAngle = Internal::circularMean(headings, weights);
117 
118  // Bundle result as tuple
119  return std::make_tuple(bestAngle, std::move(snapshots), std::move(minDifferencesOut));
120  }
