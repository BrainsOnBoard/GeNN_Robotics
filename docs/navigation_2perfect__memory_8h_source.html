87  uint8_t *snapshotMaskPtr = snapshotMask.data;
88

92  const uint8_t *end = &imageMaskPtr[imSize];
93  while(imageMaskPtr < end) {
94  // If this pixel is masked by neither of the masks
95  if (*imageMaskPtr++ != 0 && *snapshotMaskPtr++) {
