53  const std::string &capturePath)
54  {
55  // Stash capture path
56  m_CapturePath = capturePath;
57 
58  // Create socket
59  m_Socket = ::socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP);
60  if(m_Socket < 0) {
61  std::cerr << "Cannot open socket: " << strerror(errno) << std::endl;
62  return false;
63  }
64 
65  // Create socket address structure
66  memset(&m_RemoteAddress, 0, sizeof(sockaddr_in));
67  m_RemoteAddress.sin_family = AF_INET,
68  m_RemoteAddress.sin_port = htons(port),
69  m_RemoteAddress.sin_addr.s_addr = inet_addr(hostname.c_str());
70 
71  // Get initial capture packet id from time
72  // **NOTE** we intentionally cast this to 32-bit as (supposedly)
73  // Vicon Tracker struggles with large values...
74  m_CapturePacketID = (uint32_t)time(nullptr);
75  return true;
76  }

83  message << "<CaptureStart>" << std::endl;
