187  {

189  // quarter input size
190  const unsigned int inputWidth = getWidth();

192  assert(output.cols == (int) inputWidth / 2);
193  assert(output.rows == (int) inputHeight / 2);

196  // Read data and size (in bytes) from camera
197  // **NOTE** these pointers are only valid within one frame
198  void *data = nullptr;
199  uint32_t sizeBytes = 0;

201  // Check frame size is correct
202  assert(sizeBytes == (inputWidth * inputHeight * sizeof(uint16_t)));
203  const uint16_t *bayerData = reinterpret_cast<uint16_t *>(data);
204 
205  // Loop through bayer pixels
206  for (unsigned int y = 0; y < inputHeight; y += 2) {
207  // Get pointers to start of both rows of Bayer data and output
208  // RGB data
209  const uint16_t *inBG16Start = &bayerData[y * inputWidth];
210  const uint16_t *inR16Start =
211  &bayerData[((y + 1) * inputWidth) + 1];
212  uint8_t *outStart = output.ptr(y / 2);
213  for (unsigned int x = 0; x < inputWidth; x += 2) {
214  // Read Bayer pixels

217  const uint16_t r = *inR16Start;

235  // Calculates entropy, either from whole frame or within subset specified by
236  // mask
237  // **NOTE** this uses full 10-bit sensor range for calculation
238  float calculateImageEntropy(const cv::Mat &mask)
239  {
240  const unsigned int inputWidth = getWidth();
241  const unsigned int inputHeight = getHeight();
242 
243  // Check validity of mask
244  const bool noMask = (mask.cols == 0 && mask.rows == 0);
245  assert(noMask || (mask.cols == (int) (inputWidth / 2) &&
246  mask.rows == (int) (inputHeight / 2)));
247  assert(noMask || mask.type() == CV_8UC1);
248 
249  // Read data and size (in bytes) from camera
250  // **NOTE** these pointers are only valid within one frame
251  void *data = nullptr;
252  uint32_t sizeBytes = 0;
253  if (Video4LinuxCamera::capture(data, sizeBytes)) {
254  // Check frame size is correct
255  assert(sizeBytes == (inputWidth * inputHeight * sizeof(uint16_t)));
256  const uint16_t *bayerData = reinterpret_cast<uint16_t *>(data);
257 
258  // Zero a 10-bit RGB histogram for each colour channel
259  unsigned int hist[3][1024];
260  std::fill_n(hist[0], 1024, 0);
261  std::fill_n(hist[1], 1024, 0);
262  std::fill_n(hist[2], 1024, 0);
263 
264  // Loop through bayer pixels
265  unsigned int numPixels = 0;
266  for (unsigned int y = 0; y < inputHeight; y += 2) {
267  // Get pointers to start of both rows of Bayer data and output
268  // RGB data
269  const uint16_t *inBG16Start = &bayerData[y * inputWidth];
270  const uint16_t *inR16Start =
271  &bayerData[((y + 1) * inputWidth) + 1];
272  for (unsigned int x = 0; x < inputWidth; x += 2) {
273  // Read Bayer pixels
274  const uint16_t b = *(inBG16Start++);
275  const uint16_t g = *(inBG16Start++);
276  const uint16_t r = *inR16Start;
277  inR16Start += 2;
278 
279  // If no mask is in use or this pixel isn't masked
280  // **NOTE** divide by two as x and y are in terms of Bayer
281  // pixels
282  if (noMask || mask.at<uint8_t>(y / 2, x / 2) > 0) {
283  // Increment histogram bins
284  hist[0][r]++;
285  hist[1][g]++;
286  hist[2][b]++;
287 
288  // Increment totals
289  numPixels++;
290  }
291  }
292  }
293 
294  // Check pixel count
295  assert(!noMask || numPixels == (inputWidth * inputHeight / 4));
296 
297  // Sum together entropy for each colour channel
298  float entropy = 0.0f;
299  for (unsigned int c = 0; c < 3; c++) {
300  entropy -= std::accumulate(
301  std::begin(hist[c]),
302  std::end(hist[c]),
303  0.0f,
304  [numPixels](float acc, unsigned int h) {
305  if (h == 0) {
306  return acc;
307  } else {
308  const float p = (float) h / (float) numPixels;
309  return acc + (p * std::log(p));
310  }
311  });
312  }
313  return entropy;
314  } else {
315  throw std::runtime_error("Cannot read from camera");
316  }
317  }

319  // Automatically configure camera exposure and brightness to optimise image
320  // quality Use of mask inspired by "Nourani-Vatani & Roberts (2007).
321  // Automatic Camera Exposure Control." Uses image-entropy based algorithm
322  // described by "Lu, Zhang et al. (2010). Camera parameters auto-adjusting
323  // technique for robust robot vision."
324  void autoExposure(const cv::Mat &mask, int brightnessExposureConstant = 5)
325  {
326  // Configure initial brightness and exposure values
327  int32_t brightness = m_BrightnessControl.minimum;
328  int32_t exposure = std::max(m_ExposureControl.minimum,
329  brightness * brightnessExposureConstant);
330  assert(exposure < m_ExposureControl.maximum);
331 
332  // Loop until we've
333  int32_t previousBrightness = 0;
334  int32_t previousExposure = 0;
335  float previousEntropy = 0.0f;
336  while (brightness < m_BrightnessControl.maximum &&
337  exposure < m_ExposureControl.maximum) {
338  // Set exposure and brightness
339  setExposure(exposure);
340  setBrightness(brightness);
341 
342  // Wait for change to take effect
343  // **NOTE** delay found experimentally
344  std::this_thread::sleep_for(std::chrono::milliseconds(100));
345 
346  // Throw away frame so new frame is captured AFTER setting change
347  // **NOTE** this is required because of double-buffering in
348  // Video4LinuxCamera
349  void *data = nullptr;
350  uint32_t sizeBytes = 0;
351  capture(data, sizeBytes);
352 
353  // Calculate image entropy
354  const float entropy = calculateImageEntropy(mask);
355 
356  // If we have passed entropy peak
357  if (entropy < previousEntropy) {
358  std::cout << "Optimal exposure and brightness settings found: "
359  "exposure="
360  << previousExposure
361  << ", brightness=" << previousBrightness << std::endl;
362  setExposure(previousExposure);
363  setBrightness(previousBrightness);
364  return;
365  }
366  // Otherwise
367  else {
368  // Update previous state
369  previousEntropy = entropy;
370  previousBrightness = brightness;
371  previousExposure = exposure;
372 
373  // Follow path
374  // **NOTE** step size found experimentally
375  brightness += (entropy < 15.0f) ? 8 : 2;
376  exposure = (brightness * brightnessExposureConstant);
377  }
378  }
379  }
