225  void initGyro(const GyroSettings &settings)
226  {
227  // Cache gyro sensitivity
228  m_GyroSensitivity = getGyroSensitivity(settings.scale);
229 
230  // CTRL_REG1_G (Default value: 0x00)
231  // [ODR_G2][ODR_G1][ODR_G0][FS_G1][FS_G0][0][BW_G1][BW_G0]
232  // ODR_G[2:0] - Output data rate selection
233  // FS_G[1:0] - Gyroscope full-scale selection
234  // BW_G[1:0] - Gyroscope bandwidth selection
235 
236  // To disable gyro, set sample rate bits to 0. We'll only set sample
237  // rate if the gyro is enabled.
238  uint8_t ctrlReg1Value = static_cast<uint8_t>(settings.sampleRate) << 5;
239  ctrlReg1Value |= static_cast<uint8_t>(settings.scale);
240  ctrlReg1Value |= (settings.bandwidth & 0x3);
241  writeAccelGyroByte(AccelGyroReg::CTRL_REG1_G, ctrlReg1Value);
242 
243  // CTRL_REG2_G (Default value: 0x00)
244  // [0][0][0][0][INT_SEL1][INT_SEL0][OUT_SEL1][OUT_SEL0]
245  // INT_SEL[1:0] - INT selection configuration
246  // OUT_SEL[1:0] - Out selection configuration
247  writeAccelGyroByte(AccelGyroReg::CTRL_REG2_G, 0);
248 
249  // CTRL_REG3_G (Default value: 0x00)
250  // [LP_mode][HP_EN][0][0][HPCF3_G][HPCF2_G][HPCF1_G][HPCF0_G]
251  // LP_mode - Low-power mode enable (0: disabled, 1: enabled)
252  // HP_EN - HPF enable (0:disabled, 1: enabled)
253  // HPCF_G[3:0] - HPF cutoff frequency
254  uint8_t ctrlReg3Value = settings.lowPowerEnable ? (1 << 7) : 0;
255  ctrlReg3Value |= static_cast<uint8_t>(settings.hpf);
256  writeAccelGyroByte(AccelGyroReg::CTRL_REG3_G, ctrlReg3Value);
257 
258  // CTRL_REG4 (Default value: 0x38)
259  // [0][0][Zen_G][Yen_G][Xen_G][0][LIR_XL1][4D_XL1]
260  // Zen_G - Z-axis output enable (0:disable, 1:enable)
261  // Yen_G - Y-axis output enable (0:disable, 1:enable)
262  // Xen_G - X-axis output enable (0:disable, 1:enable)
263  // LIR_XL1 - Latched interrupt (0:not latched, 1:latched)
264  // 4D_XL1 - 4D option on interrupt (0:6D used, 1:4D used)
265  uint8_t ctrlReg4Value = 0;
266  if (settings.enableZ) {
267  ctrlReg4Value |= (1 << 5);
268  }
269  if (settings.enableY) {
270  ctrlReg4Value |= (1 << 4);
271  }
272  if (settings.enableX) {
273  ctrlReg4Value |= (1 << 3);
274  }
275  if (settings.latchInterrupt) {
276  ctrlReg4Value |= (1 << 1);
277  }
278  writeAccelGyroByte(AccelGyroReg::CTRL_REG4, ctrlReg4Value);
279 
280  // ORIENT_CFG_G (Default value: 0x00)
281  // [0][0][SignX_G][SignY_G][SignZ_G][Orient_2][Orient_1][Orient_0]
282  // SignX_G - Pitch axis (X) angular rate sign (0: positive, 1: negative)
283  // Orient [2:0] - Directional user orientation selection
284  uint8_t orientCfgValue = 0;
285  if (settings.flipX) {
286  orientCfgValue |= (1 << 5);
287  }
288  if (settings.flipY) {
289  orientCfgValue |= (1 << 4);
290  }
291  if (settings.flipZ) {
292  orientCfgValue |= (1 << 3);
293  }
294  writeAccelGyroByte(AccelGyroReg::ORIENT_CFG_G, orientCfgValue);
295 
296  std::cout << "Gyro initialised" << std::endl;
297  }

299  void initAccel(const AccelSettings &settings)
300  {
301  // Cache accelerometer sensitivity
302  m_AccelSensitivity = getAccelSensitivity(settings.scale);
303 
304 
305  // CTRL_REG5_XL (0x1F) (Default value: 0x38)
306  // [DEC_1][DEC_0][Zen_XL][Yen_XL][Zen_XL][0][0][0]
307  // DEC[0:1] - Decimation of accel data on OUT REG and FIFO.
308  // 00: None, 01: 2 samples, 10: 4 samples 11: 8 samples
309  // Zen_XL - Z-axis output enabled
310  // Yen_XL - Y-axis output enabled
311  // Xen_XL - X-axis output enabled
312  uint8_t ctrlReg5Value = 0;
313  if (settings.enableZ) {
314  ctrlReg5Value |= (1 << 5);
315  }
316  if (settings.enableY) {
317  ctrlReg5Value |= (1 << 4);
318  }
319  if (settings.enableX) {
320  ctrlReg5Value |= (1 << 3);
321  }
322 
323  writeAccelGyroByte(AccelGyroReg::CTRL_REG5_XL, ctrlReg5Value);
324 
325  // CTRL_REG6_XL (0x20) (Default value: 0x00)
326  // [ODR_XL2][ODR_XL1][ODR_XL0][FS1_XL][FS0_XL][BW_SCAL_ODR][BW_XL1][BW_XL0]
327  // ODR_XL[2:0] - Output data rate & power mode selection
328  // FS_XL[1:0] - Full-scale selection
329  // BW_SCAL_ODR - Bandwidth selection
330  // BW_XL[1:0] - Anti-aliasing filter bandwidth selection
331  uint8_t ctrlReg6Value = static_cast<uint8_t>(settings.sampleRate) << 5;
332  ctrlReg6Value |= static_cast<uint8_t>(settings.scale) << 3;
333  ctrlReg6Value |= static_cast<uint8_t>(settings.bandwidth);
334  writeAccelGyroByte(AccelGyroReg::CTRL_REG6_XL, ctrlReg6Value);
335 
336  // CTRL_REG7_XL (0x21) (Default value: 0x00)
337  // [HR][DCF1][DCF0][0][0][FDS][0][HPIS1]
338  // HR - High resolution mode (0: disable, 1: enable)
339  // DCF[1:0] - Digital filter cutoff frequency
340  // FDS - Filtered data selection
341  // HPIS1 - HPF enabled for interrupt function
342  writeAccelGyroByte(AccelGyroReg::CTRL_REG7_XL, static_cast<uint8_t>(settings.highResBandwidth));
343 
344  std::cout << "Accelerometer initialised" << std::endl;
345  }

348  void initMagneto(const MagnetoSettings &settings)
349  {
350  // Cache magneto sensitivity
351  m_MagnetoSensitivity = getMagnetoSensitivity(settings.scale);
352  std::cout << m_MagnetoSensitivity << std::endl;
353  // CTRL_REG1_M (Default value: 0x10)
354  // [TEMP_COMP][OM1][OM0][DO2][DO1][DO0][0][ST]
355  // TEMP_COMP - Temperature compensation
356  // OM[1:0] - X & Y axes op mode selection
357  // 00:low-power, 01:medium performance
358  // 10: high performance, 11:ultra-high performance
359  // DO[2:0] - Output data rate selection
360  // ST - Self-test enable
361  uint8_t ctrlReg1Value = 0;
362  if (settings.tempCompensationEnable) {
363  ctrlReg1Value |= (1 << 7);
364  }
365  ctrlReg1Value |= static_cast<uint8_t>(settings.xyPerformance) << 5;
366  ctrlReg1Value |= static_cast<uint8_t>(settings.sampleRate) << 2;
367  writeMagnetoByte(MagnetoReg::CTRL_REG1, ctrlReg1Value);
368 
369  // CTRL_REG2_M (Default value 0x00)
370  // [0][FS1][FS0][0][REBOOT][SOFT_RST][0][0]
371  // FS[1:0] - Full-scale configuration
372  // REBOOT - Reboot memory content (0:normal, 1:reboot)
373  // SOFT_RST - Reset config and user registers (0:default, 1:reset)
374  const uint8_t ctrlReg2Value = static_cast<uint8_t>(settings.scale) << 5;
375  writeMagnetoByte(MagnetoReg::CTRL_REG2, ctrlReg2Value);
376 
377  // CTRL_REG3_M (Default value: 0x03)
378  // [I2C_DISABLE][0][LP][0][0][SIM][MD1][MD0]
379  // I2C_DISABLE - Disable I2C interace (0:enable, 1:disable)
380  // LP - Low-power mode cofiguration (1:enable)
381  // SIM - SPI mode selection (0:write-only, 1:read/write enable)
382  // MD[1:0] - Operating mode
383  // 00:continuous conversion, 01:single-conversion,
384  // 10,11: Power-down
385  uint8_t ctrlReg3Value = 0;
386  if (settings.lowPowerEnable) {
387  ctrlReg3Value |= (1 << 5);
388  }
389  ctrlReg3Value |= static_cast<uint8_t>(settings.operatingMode);
390  writeMagnetoByte(MagnetoReg::CTRL_REG3, ctrlReg3Value);
391 
392  // CTRL_REG4_M (Default value: 0x00)
393  // [0][0][0][0][OMZ1][OMZ0][BLE][0]
394  // OMZ[1:0] - Z-axis operative mode selection
395  // 00:low-power mode, 01:medium performance
396  // 10:high performance, 10:ultra-high performance
397  // BLE - Big/little endian data
398  const uint8_t ctrlReg4Value = static_cast<uint8_t>(settings.zPerformance) << 2;
399  writeMagnetoByte(MagnetoReg::CTRL_REG4, ctrlReg4Value);
400 
401  // CTRL_REG5_M (Default value: 0x00)
402  // [0][BDU][0][0][0][0][0][0]
403  // BDU - Block data update for magnetic data
404  // 0:continuous, 1:not updated until MSB/LSB are read
405  writeMagnetoByte(MagnetoReg::CTRL_REG5, 0);
406 
407  std::cout << "Magnetometer initialised" << std::endl;
408  }

410  void calibrateAccelGyro()
411  {
412  std::cout << "Calibrating accelerometer and gyroscope" << std::endl;
413 
414  setFIFOEnabled(true);
415  setFIFOMode(FIFOMode::Threshold, 31);
416 
417  // Accumulate 32 samples
418  unsigned int numSamples = 0;
419  while(numSamples < 31) {
420  numSamples = getNumFIFOSamples();
421  }
422 
423  // Accumulate bias from sensor samples
424  // **NOTE** 32-bit to prevent overflow
425  int32_t accelBias[3] = {0, 0, 0};
426  int32_t gyroBias[3] = {0, 0, 0};
427  for(unsigned int i = 0; i < numSamples; i++)
428  {
429  // Read a sample from gyroscope
430  int16_t gyroSample[3];
431  readGyro(gyroSample);
432 
433  // Add to gyro bias
434  gyroBias[0] += gyroSample[0];
435  gyroBias[1] += gyroSample[1];
436  gyroBias[2] += gyroSample[2];
437 
438  // Read a sample from accelerometer
439  int16_t accelSample[3];
440  readAccel(accelSample);
441 
442  // Add to acclerometer bias
443  // **NOTE** we subtract gravity from Y as sensor is vertical in current robot
444  accelBias[0] += accelSample[0];
445  accelBias[1] += accelSample[1] - (int16_t)(1.0f / m_AccelSensitivity);
446  accelBias[2] += accelSample[2];
447  }
448 
449  // Divide biases by number of samples to get means
450  std::transform(std::begin(accelBias), std::end(accelBias), std::begin(m_AccelBias),
451  [numSamples](int32_t v){ return v / (int32_t)numSamples; });
452  std::transform(std::begin(gyroBias), std::end(gyroBias), std::begin(m_GyroBias),
453  [numSamples](int32_t v){ return v / (int32_t)numSamples; });
454 
455  setFIFOEnabled(false);
456  setFIFOMode(FIFOMode::Off, 0);
457 
458  std::cout << "\tAccel bias:" << m_AccelBias[0] << "," << m_AccelBias[1] << "," << m_AccelBias[2] << std::endl;
459  std::cout << "\tGyro bias:" << m_GyroBias[0] << "," << m_GyroBias[1] << "," << m_GyroBias[2] << std::endl;
