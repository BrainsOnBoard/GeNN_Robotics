93  void create(const cv::Size &cameraResolution,
94  const cv::Size &unwrappedResolution,
95  double centreX = 0.5,
96  double centreY = 0.5,
97  double inner = 0.1,
98  double outer = 0.5,
99  int offsetDegrees = 0,
100  bool flip = false)
101  {
102  // convert relative (0.0 to 1.0) to absolute pixel values
103  m_CentrePixel = cv::Point(
104  (int) round((double) cameraResolution.width * centreX),
105  (int) round((double) cameraResolution.height * centreY));
106  m_InnerPixel = (int) round((double) cameraResolution.height * inner);
107  m_OuterPixel = (int) round((double) cameraResolution.height * outer);
108 
109  // Save params
110  m_CameraResolution = cameraResolution;
111  m_UnwrappedResolution = unwrappedResolution;
112  m_OffsetDegrees = offsetDegrees;
113  m_Flip = flip;
114 
115  // Build unwrap maps
116  createMaps();
117  }

119  void updateMaps()
120  {
121  // Build unwrap maps
122  const float pi = 3.141592653589793238462643383279502884f;
123  float offsetRadians = m_OffsetDegrees * pi / 180.0f;
124  for (int i = 0; i < m_UnwrappedResolution.height; i++) {
125  for (int j = 0; j < m_UnwrappedResolution.width; j++) {
126  // Get i as a fraction of unwrapped height, flipping if desired
127  const float iFrac =
128  m_Flip ? 1.0f - ((float) i /
129  (float) m_UnwrappedResolution.height)
130  : ((float) i /
131  (float) m_UnwrappedResolution.height);
132 
133  // Convert i and j to polar
134  const float r =
135  iFrac * (m_OuterPixel - m_InnerPixel) + m_InnerPixel;
136  const float th =
137  (((float) j / (float) m_UnwrappedResolution.width) *
138  2.0f * pi) +
139  offsetRadians;
140 
141  // Remap onto sphere
142  const float x = m_CentrePixel.x - r * sin(th);
143  const float y = m_CentrePixel.y + r * cos(th);
144  m_UnwrapMapX.at<float>(i, j) = x;
145  m_UnwrapMapY.at<float>(i, j) = y;
146  }
147  }
148  }
