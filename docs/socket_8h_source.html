117  void read(void *buffer, size_t len)
118  {
119  std::lock_guard<std::mutex> guard(m_ReadMutex);
120  checkSocket();
121 
122  // initially, copy over any leftover bytes in m_Buffer
123  size_t start = 0;
124  if (m_BufferBytes > 0) {
125  size_t tocopy = std::min(len, m_BufferBytes);
126  memcpy(buffer, &m_Buffer[m_BufferStart], tocopy);
127  start += tocopy;
128  len -= tocopy;
129  debitBytes(tocopy);
130  }
131 
132  // keep reading from socket until we have enough bytes
133  while (len > 0) {
134  size_t nbytes = readOnce((char *) buffer, start, len);
135  start += nbytes;
136  len -= nbytes;
137  }
138  }

141  std::string readLine()
142  {
143  std::lock_guard<std::mutex> guard(m_ReadMutex);
144  checkSocket();
145 
146  std::ostringstream oss;
147  while (true) {
148  if (m_BufferBytes == 0) {
149  m_BufferBytes += readOnce(m_Buffer.data(),
150  m_BufferStart,
151  DefaultBufferSize - m_BufferStart);
152  }
153 
154  // look for newline char
155  for (size_t i = 0; i < m_BufferBytes; i++) {
156  char &c = m_Buffer[m_BufferStart + i];
157  if (c == '\n') {
158  c = '\0';
159  oss << std::string(&m_Buffer[m_BufferStart]);
160  debitBytes(i + 1);
161 
162  std::string outstring = oss.str();
163  if (m_Print) {
164  std::cout << "<<< " << outstring << std::endl;
165  }
166  return outstring;
167  }
168  }
169 
170  // if newline is not present, append the text we received and try
171  // another read
172  oss << std::string(&m_Buffer[m_BufferStart], m_BufferBytes);
173  debitBytes(m_BufferBytes);
174  }
175  }
