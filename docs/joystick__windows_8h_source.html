108 
109  // unset Pressed and Released bits for buttons
110  for (auto &s : m_ButtonState) {
111  s &= StateDown;
112  }
113 
114  // Check that something has changed
115  if (m_State.dwPacketNumber == m_NewState.dwPacketNumber) {
116  return false;
117  }
118 
119  // Check buttons for changes
120  for (size_t i = toIndex(JButton::Start); i < toIndex(JButton::LENGTH); i++) {
121  uint8_t &s = m_ButtonState[i];
122  const uint8_t pressed = (m_NewState.Gamepad.wButtons >> i) & 1;
123  if (pressed != (s & StateDown)) {
124  // ... then this button has been pressed or released
125  if (pressed) {
126  s |= (StateDown | StatePressed); // set StateDown and StatePressed
127  } else {
128  s &= ~StateDown; // clear StateDown
129  s |= StateReleased; // set StateReleased
130  }
131

134  }
135  }
